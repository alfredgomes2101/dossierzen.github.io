function navigate(target) {
  if (target === currentScreen) return;
  screenHistory.push(currentScreen);
  const ci = SCREEN_ORDER.indexOf(currentScreen);
  const ni = SCREEN_ORDER.indexOf(target);
  const dir = ni >= ci ? 'slide-left' : 'slide-right';
  const el = document.getElementById('screen-' + currentScreen);
  el.classList.remove('active');
  el.classList.add(dir);
  const newEl = document.getElementById('screen-' + target);
  newEl.classList.remove('slide-left','slide-right');
  newEl.classList.add('active');
  currentScreen = target;
  updateNav(target);
  if (target === 'vault') renderCatGrid();
}

function navigateBack() {
  if (screenHistory.length > 0) {
    const prev = screenHistory.pop();
    const ci = SCREEN_ORDER.indexOf(currentScreen);
    const pi = SCREEN_ORDER.indexOf(prev);
    const el = document.getElementById('screen-' + currentScreen);
    el.classList.remove('active');
    el.classList.add('slide-right');
    const prevEl = document.getElementById('screen-' + prev);
    prevEl.classList.remove('slide-left','slide-right');
    prevEl.classList.add('active');
    currentScreen = prev;
    updateNav(prev);
  }
}

function updateNav(screen) {
  const navMap = {home:'home',vault:'vault',upload:'vault',dossier:'dossier',ai:'ai',profile:'profile'};
  document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
  const navId = 'nav-' + (navMap[screen] || screen);
  const navEl = document.getElementById(navId);
  if (navEl) navEl.classList.add('active');
}

    const bg = s==='ok'?'background:rgba(46,204,138,.12);color:#2ecc8a':s==='exp'?'background:rgba(224,82,82,.12);color:#e05252':s==='soon'?'background:rgba(240,120,64,.12);color:#f07840':'background:rgba(74,82,120,.1);color:#4a5278';
    const col = colorMap[cat]||'di-blue';
    const expLine = d.expiry ? `Exp: ${d.expiry}` : (d.meta||'');
    return `
      <div class="doc-full-item">
        <div class="dfi-main">
          <div class="dfi-icon ${col}">${d.icon}</div>
          <div class="dfi-info">
            <div class="dfi-name">${d.name}</div>
            <div class="dfi-sub">${expLine}</div>
            <span class="dfi-expiry" style="${bg}">${badgeTxt}</span>
          </div>
          <div class="dfi-actions">
            ${s==='miss'?`<button class="dfi-btn primary" onclick="navigate('upload')">＋ Ajouter</button>`:`<button class="dfi-btn" onclick="showToast('👁 Aperçu du document')">👁</button><button class="dfi-btn" onclick="showToast('↑ Téléchargé')">↑</button>`}
          </div>
        </div>
      </div>`;
  }).join('');
}

function filterVaultSearch(val) {
  if (!val) { renderCatGrid(); return; }
  showAllCats();
  const grid = document.getElementById('cat-grid');
  let html = '';
  Object.entries(CAT_META).forEach(([key, cat]) => {
    const docs = DOC_DATA[key] || [];
    const match = docs.filter(d => d.name.toLowerCase().includes(val.toLowerCase()));
    if (match.length > 0) {
      html += `<div class="cat-ov-card" onclick="showCatDocs('${key}')">
        <div class="coc-top"><div class="coc-icon-wrap ${cat.color}">${cat.icon}</div><div class="coc-count-chip">${match.length}</div></div>
        <div class="coc-name">${cat.name_fr}</div>
        <div class="coc-hint">${match.map(d=>d.name).join(', ')}</div></div>`;
    }
  });
  grid.innerHTML = html || '<div style="padding:20px;color:var(--text2);text-align:center">Aucun résultat</div>';
}

function saveDocument() {
  const name = document.getElementById('doc-name-input').value.trim();
  const cat = selectedCatUpload;
  if (!name) { showToast('⚠️ Veuillez nommer le document'); return; }
  if (!cat) { showToast('⚠️ Veuillez choisir une catégorie'); return; }
  if (!uploadedDocs[cat]) uploadedDocs[cat] = [];
  uploadedDocs[cat].push({icon:'📄',name,expiry:null,status:'ok',meta:'Ajouté par vous'});
  showToast('✅ Document enregistré dans le coffre !');
  // reset form
  setTimeout(() => {
    document.getElementById('doc-name-input').value = '';
    document.getElementById('ocr-result').style.display = 'none';
    document.querySelectorAll('.cp-item').forEach(i => i.classList.remove('selected'));
    selectedCatUpload = null;
    document.getElementById('drop-zone').innerHTML = `
      <div id="dz-idle">
        <span class="dz-icon">📷</span>
        <div class="dz-text" data-i18n="dz_text">Prendre une photo ou glisser-déposer</div>
        <div class="dz-hint" data-i18n="dz_hint">JPG · PNG · PDF · Max 10MB</div>
        <button class="dz-btn" onclick="triggerFileInput()" data-i18n="dz_btn">Choisir un fichier</button>

        <div class="dz-spinner"></div>
        <span>Analyse OCR en cours...</span>
      </div>`;

  }, 800);
}

function sendQuickQ(el) {
  const text = el.textContent.replace('💬 ', '');
  addBubble(text, 'usr');
  el.remove();
  setTimeout(() => {
    addBubble(LANG[currentLang].ai_response || LANG.fr.ai_response, 'bot');
  }, 900);
}

function sendAiMsg() {
  const inp = document.getElementById('ai-input');
  const text = inp.value.trim();
  if (!text) return;
  addBubble(text, 'usr');
  inp.value = '';
  inp.style.height = '';
  setTimeout(() => {
    addBubble(LANG[currentLang].ai_response || LANG.fr.ai_response, 'bot');
  }, 900);
}

function handleAiKey(e) {
  if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendAiMsg(); }
}

function autoResize(ta) {
  ta.style.height = 'auto';
  ta.style.height = Math.min(ta.scrollHeight, 90) + 'px';
}

function addBubble(text, role) {
  const msgs = document.getElementById('ai-messages');
  const wrap = document.createElement('div');
  wrap.className = 'msg-wrap-' + role;
  const now = new Date();
  const time = now.getHours().toString().padStart(2,'0') + ':' + now.getMinutes().toString().padStart(2,'0');
  wrap.innerHTML = `<div class="bubble bubble-${role}">${text}</div><div class="msg-time" style="${role==='usr'?'text-align:right':''}">${time}</div>`;
  msgs.appendChild(wrap);
  msgs.scrollTop = msgs.scrollHeight;
}

/* ═══════════════════════════════════════════
   I18N ENGINE
═══════════════════════════════════════════ */
const langNames = {fr:'Français',ar:'العربية',en:'English',pt:'Português',es:'Español',bn:'বাংলা',hi:'हिन्दी',ta:'தமிழ்',tr:'Türkçe',it:'Italiano'};
const rtlLangs = ['ar'];

function setLang(code) {
  currentLang = code;
  const L = LANG[code] || LANG.fr;
  document.documentElement.lang = code;
  document.documentElement.dir = rtlLangs.includes(code) ? 'rtl' : 'ltr';

  // highlight button
  document.querySelectorAll('.lang-pill').forEach(b => {
    b.classList.toggle('active', b.getAttribute('onclick').includes("'"+code+"'"));
  });

  // apply text nodes
  document.querySelectorAll('[data-i18n]').forEach(el => {
    const key = el.getAttribute('data-i18n');
    if (L[key] !== undefined) el.innerHTML = L[key];
  });

  // chips
  document.querySelectorAll('.chip[data-i18n-chip]').forEach(chip => {
    const key = 'chip_' + chip.dataset.cat;
    if (L[key]) chip.innerHTML = L[key];
  });

  // placeholders
  const aiInp = document.getElementById('ai-input');
  if (aiInp) aiInp.placeholder = L.ai_placeholder || 'Posez votre question...';
  const vsearch = document.getElementById('vault-search-input');
  if (vsearch) vsearch.placeholder = L.vault_search || 'Rechercher...';

  // profile lang display
  const ld = document.getElementById('current-lang-display');
  if (ld) ld.textContent = langNames[code] || code;

  // re-render vault if visible
  if (currentScreen === 'vault') renderCatGrid();
}
